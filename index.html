// ફક્ત Admin જ પોસ્ટ કરી શકે તેવો સુધારેલો ફંકશન async function postNotice() { // પહેલા ચેક કરો કે લોગીન કરનાર Admin છે કે નહીં if (role !== "admin") { alert("માત્ર એડમિન જ નોટિસ મૂકી શકે છે!"); return; } const msg = document.getElementById('noticeMsg').value.trim(); const fileInput = document.getElementById('noticeFile'); let fileData = null; let fileType = ""; if(!msg && fileInput.files.length === 0) { alert("કંઈક લખો અથવા ફાઈલ પસંદ કરો"); return; } // જો ફાઈલ સિલેક્ટ કરી હોય તો તેને Base64 માં બદલો (WhatsApp જેવું ફીચર) if(fileInput.files.length > 0) { const file = fileInput.files[0]; fileType = file.type; // ફાઈલ સાઈઝ ચેક (વધારે મોટી ફાઈલ ડેટાબેઝ ને સ્લો કરી શકે) if (file.size > 2 * 1024 * 1024) { alert("ફાઈલ 2MB થી નાની હોવી જોઈએ"); return; } fileData = await convertBase64(file); } const newNotice = { message: msg, file: fileData, type: fileType, time: new Date().toLocaleString('gu-IN'), // ગુજરાતી સમય મુજબ sender: "ADMIN" }; // Firebase માં ડેટા મોકલો noticeDB.push(newNotice).then(() => { alert("નોટિસ સફળતાપૂર્વક શેર કરવામાં આવી!"); document.getElementById('noticeMsg').value = ""; document.getElementById('noticeFile').value = ""; document.getElementById('fileName').innerText = ""; }).catch((error) => { alert("ભૂલ આવી: " + error.message); }); } // ફાઈલને ડેટા સ્ટ્રીમમાં બદલવા માટેનું ફંકશન function convertBase64(file) { return new Promise((resolve, reject) => { const reader = new FileReader(); reader.readAsDataURL(file); reader.onload = () => resolve(reader.result); reader.onerror = error => reject(error); }); }
